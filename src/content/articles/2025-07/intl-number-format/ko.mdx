import { formatKoreanCurrency } from "../../utils/number";

describe("formatKoreanCurrency", () => {
  test.each([
    [999, "999"],
    [5000, "5,000"],
    [9999, "9,999"],
    [10000, "1.0만"],
    [76870448, "7687.0만"],
    [100000000, "1.0억"],
    [532000000, "5.3억"],
    [10000000000, "100.0억"],
    [12343232000, "123.4억"],
    [1000000000000, "1.0조"],
    [2500000000000, "2.5조"],
  ])("formats %s to %s", (input, expected) => {
    expect(formatKoreanCurrency(input)).toBe(expected);
  });
});
